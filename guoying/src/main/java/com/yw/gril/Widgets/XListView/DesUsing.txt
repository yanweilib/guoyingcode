说明：
    上拉加载，下拉刷新

使用：
    layout：
        <XListView
            android:id="@+id/lv_content"
            android:layout_width="match_parent"
            android:layout_height="match_parent" >
        </XListView>

     activity：

            implements XListView.IXListViewListener

            public XListView xListView;
            private String upRefreshTime;

             xListView = (XListView) view.findViewById(R.id.list_lv);
             xListView.setXListViewListener(this);
             xListView.setPullRefreshEnable(true);
             xListView.setPullLoadEnable(true);


        //刷新listview数据
        qualifyingAdapter.notifyDataSetChanged();
        xListView.invalidate();

                 /**
                  * 刷新完毕
                  */
                 private void completedRefresh() {
                     handler.postDelayed(new Runnable() {
                         @Override
                         public void run() {
                             xListView.stopRefresh();
                             xListView.stopLoadMore();
                         }
                     },300);
                     upRefreshTime = CommentHelper.currentTime();
                     xListView.setRefreshTime(upRefreshTime);
                 }
